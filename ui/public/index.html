Quick Example

const store = createEventStore({
  storage: new SqliteStorage('./events.db'),
  secret: 'your-hmac-secret',
  consistency: {
    eventTypes: {
      StudentSubscribed: {
        keys: [
          { name: 'course', path: 'courseId' },
          { name: 'student', path: 'studentId' }
        ]
      }
    }
  }
});

// Read with DCB query
const { events, token } = await store.read({
  conditions: [
    { type: 'StudentSubscribed', key: 'course', value: 'cs101' }
  ]
});

// Append with consistency check
const result = await store.append([
  { type: 'StudentSubscribed', data: { courseId: 'cs101', studentId: 'alice' } }
], token);

if (isConflict(result)) {
  // result.conflictingEvents shows what changed
  // result.newToken for retry
}

Architecture

events (append-only log) │ ├── position, event_type, data, timestamp │ event_keys (consistency index) │ ├── position, key_name, key_value │ metadata │ └── config_hash → auto-reindex on config change
